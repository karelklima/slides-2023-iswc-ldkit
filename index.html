1. Create data schema

      import { dbo, rdfs, xsd } from "ldkit/namespaces";

      const PersonSchema = {
        "@type": dbo.Person,
        name: rdfs.label,
        abstract: dbo.abstract,
        birthDate: {
          "@id": dbo.birthDate,
          "@type": xsd.date,
        },
      } as const;
